### `src/lib/ocr/tesseract-worker.ts`
Wrapper around Tesseract.js. Manages worker lifecycle.

```ts
import { createWorker, Worker } from 'tesseract.js';

let worker: Worker | null = null;

export async function initOcrWorker(onProgress?: (p: number) => void): Promise<void> {
  if (worker) return;
  worker = await createWorker('eng', 1, {
    logger: (m) => {
      if (m.status === 'recognizing text') onProgress?.(Math.round(m.progress * 100));
    },
  });
}

export async function recognizeText(imageData: string): Promise<string> {
  if (!worker) throw new Error('OCR worker not initialized');
  const { data: { text } } = await worker.recognize(imageData);
  return text.trim();
}

export async function terminateOcrWorker(): Promise<void> {
  if (worker) { await worker.terminate(); worker = null; }
}
```

Pre-load on scan page mount (ADR-010). Show progress indicator during first load (~15MB download).

### `src/lib/ocr/captioning-pipeline.ts`
Wrapper around @huggingface/transformers for image captioning.

```ts
import { pipeline, env } from '@huggingface/transformers';

// Use browser cache (IndexedDB) for model files (ADR-013)
env.allowLocalModels = false;
env.useBrowserCache = true;

let captioner: any = null;

export async function initCaptioner(onProgress?: (p: number) => void): Promise<void> {
  if (captioner) return;
  captioner = await pipeline('image-to-text', 'Xenova/vit-gpt2-image-captioning', {
    progress_callback: (progress: any) => {
      if (progress.status === 'progress') {
        onProgress?.(Math.round(progress.progress));
      }
    },
  });
}

export async function captionImage(imageData: string): Promise<string> {
  if (!captioner) throw new Error('Captioner not initialized');
  const result = await captioner(imageData);
  return result[0]?.generated_text?.trim() || '';
}
```

```tsx
  const { data: { user } } = await supabase.auth.getUser();

  let query = supabase
    .from('collection_items')
    .select('*', { count: 'exact' })
    .eq('user_id', user.id)
    .order('created_at', { ascending: false });

  // Apply filters from searchParams
  if (searchParams.genre) query = query.contains('genres', [searchParams.genre]);
  if (searchParams.yearFrom) query = query.gte('year', searchParams.yearFrom);
  if (searchParams.yearTo) query = query.lte('year', searchParams.yearTo);
  if (searchParams.condition) query = query.eq('condition_media', searchParams.condition);
  if (searchParams.search) query = query.or(`title.ilike.%${searchParams.search}%,artist.ilike.%${searchParams.search}%`);

  // Pagination
  const page = Number(searchParams.page) || 1;
  const limit = 24;
  query = query.range((page - 1) * limit, page * limit - 1);

  const { data: items, count } = await query;

  // Get unique genres for filter dropdown
  const { data: genresRaw } = await supabase
    .from('collection_items')
    .select('genres')
    .eq('user_id', user.id);
  const genres = [...new Set(genresRaw?.flatMap(r => r.genres) || [])].sort();

  // Map snake_case rows to VinylCardProps
  const cards: VinylCardProps[] = items.map(mapToVinylCard);

  return (
    <div className="p-4 md:p-6 space-y-6">
      <h1>My Collection</h1>
      <FilterBar genres={genres} ... />       {/* Client component */}
      <VinylGrid items={cards} emptyMessage="No records yet. Scan or search to add your first vinyl!" />
      <Pagination currentPage={page} totalPages={Math.ceil((count || 0) / limit)} />
    </div>
  );
}
```

FilterBar updates URL searchParams (via `router.push` with query string) so the Server Component refetches.

- **Client Components ("use client"):** auth-form.tsx, mobile-nav.tsx, user-menu.tsx, filter-bar.tsx, discogs-result-card.tsx, scan/page.tsx + all scan components, search/page.tsx + search components, vinyl-detail.tsx (has edit/delete buttons), all dialogs, distribution-chart.tsx

### Image Optimization
- Configure `next.config.js` `images.remotePatterns` for `i.discogs.com`
- Use `next/image` everywhere with `fill` + `object-cover` for consistent aspect ratios
- Placeholder: blurred vinyl disc SVG for missing covers

### Mobile-First (ADR-007)
- All layouts start with mobile dimensions, scale up with Tailwind breakpoints
- Touch targets minimum 44x44px
- Bottom navigation uses safe-area insets: `pb-[env(safe-area-inset-bottom)]`
- Scan button is prominent center tab on mobile

### Performance
- Tesseract.js worker + Transformers.js pipeline lazy-loaded only on `/scan` page visit (ADR-010, ADR-013)
- Show download progress with estimated sizes on first load
- Use `dynamic(() => import(...), { ssr: false })` for scan components to prevent server-side import of browser-only libraries
- Collection/wishlist pages are Server Components — no client JS for initial render
- FilterBar updates URL searchParams to leverage Next.js streaming + server data fetching

### Accessibility
- All interactive elements have visible focus rings (amber ring color from theme)
- Form fields have associated `<Label>` elements
- Dialog components use Radix primitives (via shadcn/ui) with proper ARIA
- Color-coded condition badges also include text labels (not color-only)
- Camera capture has file upload fallback for accessibility
